import java.io.BufferedReader;
import java.io.FileReader;
import java.io.IOException;
import java.util.ArrayList;
import java.util.List;

public class LoadFile {
    public static void main(String[] args) {
        // Replace with the path to your file
        String filePath = "C:\\Users\\admin\\Desktop\\School\\Coding\\ENGG1420\\Final_project\\CoursesDatabase.txt";

        // Using BufferedReader and FileReader to read file into 2D list
        List<List<String>> data = FileComprehension(filePath);

        // Print the 2D list
        System.out.println("Content of the 2D list:");
        for (List<String> row : data) {
            System.out.println(row);
        }
        System.out.println(data.get(0));
    }

    public static List<List<String>> FileComprehension(String filePath) {
        List<List<String>> data = new ArrayList<>();
        try (BufferedReader br = new BufferedReader(new FileReader(filePath))) {
            String line;
            while ((line = br.readLine()) != null) {
                // Split each line by a delimiter, e.g., tab
                String[] parts = line.split(",");
                List<String> row = new ArrayList<>();
                for (String part : parts) {
                    row.add(part.trim());
                }
                data.add(row);
            }
        } catch (IOException e) {
            e.printStackTrace();
        }
        return data;
    }
}


public static void main(String[] args) {
        LoadFile system = new LoadFile();
        String[][] arr = system.readFromFile("C:\\Users\\dalyc\\Desktop\\Student.txt\\");
        System.out.println(arr[10][11]);
